Напиши функцію конструктор Storage який створює об'єкти Для управління складом товарів. При виклику отримуватиме один агрумент - початковий масив товарів, і записувати їх у властивість items. Додай методи класу: getItems() - повертайте масив товарів addItems(item) - отримує новий товар та додає його до поточних removeItem(item) - отримує товар і, якщо він є, видаляє його з поточних

const arr = ["apple", "banana", "mango"]

JS

      
        class Storage {
        constructor(items) {
          this.items = items;
        }
        getItems() {
          return this.items;
        }
        addItems(item) {
          this.items.push(item);
        }
        removeItem(item) {
          this.items = this.items.filter((i) => i !== item);
        }
      }
        
      const arr = ["apple", "banana", "mango"];

      const storage = new Storage(arr);

      console.log(storage.getItems());

      storage.addItems("orange");
      console.log(storage.getItems()); 

      storage.removeItem("banana");
      console.log(storage.getItems());
