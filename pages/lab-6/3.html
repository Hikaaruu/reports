        document.getElementById("shrinkButton").addEventListener("click", shrinkSquare);
        document.getElementById("growButton").addEventListener("click", growSquare);
        
        let squareSize = 200; // Initial size of the square
        
        function shrinkSquare() {
          squareSize -= 15;
          document.getElementById("square").style.width = squareSize + "px";
          document.getElementById("square").style.height = squareSize + "px";
        }
        
        function growSquare() {
          squareSize += 15;
          document.getElementById("square").style.width = squareSize + "px";
          document.getElementById("square").style.height = squareSize + "px";
        }
