JS

      
        document.getElementById("createButton").addEventListener("click", function() {
          var inputAmount = document.getElementById("inputAmount").value;
          
          // Validate input
          if (inputAmount < 1 || inputAmount > 100) {
            alert("Please enter a number between 1 and 100.");
            return;
          }
        
          // Clear previous boxes
          destroyBoxes();
        
          // Create new boxes
          createBoxes(inputAmount);
        });
        
        document.getElementById("destroyButton").addEventListener("click", function() {
          destroyBoxes();
        });
        
        function createBoxes(amount) {
          var boxesContainer = document.getElementById("boxes");
          for (var i = 0; i < amount; i++) {
            var box = document.createElement("div");
            box.classList.add("box");
            box.style.width = 30 + i * 10 + "px";
            box.style.height = 30 + i * 10 + "px";
            box.style.backgroundColor = getRandomHexColor();
            boxesContainer.appendChild(box);
          }
        }
        
        function destroyBoxes() {
          var boxesContainer = document.getElementById("boxes");
          boxesContainer.innerHTML = "";
        }
        
        function getRandomHexColor() {
          return `#${Math.floor(Math.random() * 16777215).toString(16).padStart(6, "0")}`;
        }
